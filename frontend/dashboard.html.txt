  <script>
    /**
     * 仪表盘主控制器
     * 负责面板切换、模块初始化、自动刷新
     */
    ;(function () {
      'use strict';

      // ---------- 模块注册 ----------
      var modules = {
        overview:    DashOverview,
        production:  DashProduction,
        maintenance: DashMaintenance,
        faults:      DashFaults,
        machines:    DashMachines,
      };

      // ---------- 初始化所有模块 ----------
      Object.keys(modules).forEach(function (name) {
        if (modules[name] && typeof modules[name].init === 'function') {
          modules[name].init();
        }
      });

      // ---------- 侧边栏切换 ----------
      var navItems = document.querySelectorAll('.nav-item');
      var panels = document.querySelectorAll('.panel');

      navItems.forEach(function (btn) {
        btn.addEventListener('click', function () {
          var target = btn.getAttribute('data-panel');
          switchPanel(target);
        });
      });

      function switchPanel(name) {
        // 导航高亮
        navItems.forEach(function (b) { b.classList.remove('active'); });
        document.querySelector('.nav-item[data-panel="' + name + '"]').classList.add('active');

        // 面板显示
        panels.forEach(function (p) { p.classList.remove('active'); });
        document.getElementById('panel-' + name).classList.add('active');

        // 切换时刷新
        if (modules[name] && typeof modules[name].refresh === 'function') {
          modules[name].refresh();
        }
      }

      // ---------- 总览面板自动刷新 ----------
      setInterval(function () {
        var overviewPanel = document.getElementById('panel-overview');
        if (overviewPanel.classList.contains('active')) {
          modules.overview.refresh();
        }
      }, 30000);

      // ---------- Toast 全局方法 ----------
      window.showToast = function (message, type) {
        type = type || 'success';
        var container = document.getElementById('toastContainer');
        var toast = document.createElement('div');
        toast.className = 'toast toast-' + type;
        toast.textContent = message;
        container.appendChild(toast);
        setTimeout(function () {
          toast.classList.add('toast-out');
          toast.addEventListener('animationend', function () { toast.remove(); });
        }, 2500);
      };
	  
	      // ---------- 全局弹窗工具 ----------
    window.Modal = (function () {
      var editOverlay = document.getElementById('editModal');
      var editTitle = document.getElementById('editModalTitle');
      var editBody = document.getElementById('editModalBody');
      var editConfirm = document.getElementById('editConfirmBtn');
      var editCancel = document.getElementById('editCancelBtn');
      var editClose = document.getElementById('editModalClose');

      var delOverlay = document.getElementById('deleteModal');
      var delText = document.getElementById('deleteModalText');
      var delConfirm = document.getElementById('deleteConfirmBtn');
      var delCancel = document.getElementById('deleteCancelBtn');
      var delClose = document.getElementById('deleteModalClose');

      var onEditConfirm = null;
      var onDeleteConfirm = null;

      // 编辑弹窗
      function openEdit(title, bodyHTML, onConfirm) {
        editTitle.textContent = title;
        editBody.innerHTML = bodyHTML;
        onEditConfirm = onConfirm;
        editOverlay.classList.add('open');
      }
      function closeEdit() {
        editOverlay.classList.remove('open');
        onEditConfirm = null;
      }
      editClose.addEventListener('click', closeEdit);
      editCancel.addEventListener('click', closeEdit);
      editOverlay.addEventListener('click', function (e) { if (e.target === editOverlay) closeEdit(); });
      editConfirm.addEventListener('click', function () {
        if (onEditConfirm) onEditConfirm(closeEdit);
      });

      // 删除确认弹窗
      function openDelete(text, onConfirm) {
        delText.textContent = text || '确定要删除这条记录吗？此操作不可撤销。';
        onDeleteConfirm = onConfirm;
        delOverlay.classList.add('open');
      }
      function closeDelete() {
        delOverlay.classList.remove('open');
        onDeleteConfirm = null;
      }
      delClose.addEventListener('click', closeDelete);
      delCancel.addEventListener('click', closeDelete);
      delOverlay.addEventListener('click', function (e) { if (e.target === delOverlay) closeDelete(); });
      delConfirm.addEventListener('click', function () {
        if (onDeleteConfirm) onDeleteConfirm(closeDelete);
      });

      return { openEdit: openEdit, closeEdit: closeEdit, openDelete: openDelete, closeDelete: closeDelete };
    })();


    })();
  </script>
